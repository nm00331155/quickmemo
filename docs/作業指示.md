Copyclass ModelManager(private val context: Context) {

    companion object {
        private const val MODEL_FILENAME = "qwen3-1.7b-q4_k_m.gguf"
        private const val MODEL_URL = "https://huggingface.co/Qwen/Qwen3-1.7B-GGUF/resolve/main/qwen3-1.7b-q4_k_m.gguf"
        private const val MODEL_SIZE_BYTES = 1_100_000_000L  // 約1.1GB
        private const val PREFS_KEY_MODEL_SHA256 = "model_sha256"
        // SHA256は実際のファイルに合わせて後で更新する
        private const val EXPECTED_SHA256 = "TODO_REPLACE_WITH_ACTUAL_HASH"
    }

    // モデルファイルの保存先: 内部ストレージ
    private val modelDir = File(context.filesDir, "ai_models")
    private val modelFile = File(modelDir, MODEL_FILENAME)

    enum class ModelStatus {
        NOT_DOWNLOADED,     // 未ダウンロード
        DOWNLOADING,        // ダウンロード中
        DOWNLOADED,         // ダウンロード済み・利用可能
        DOWNLOAD_FAILED,    // ダウンロード失敗
        CORRUPTED           // ファイル破損（再ダウンロード必要）
    }

    // 現在の状態を StateFlow で公開
    private val _status = MutableStateFlow(checkCurrentStatus())
    val status: StateFlow<ModelStatus> = _status.asStateFlow()

    // ダウンロード進捗 (0.0 ~ 1.0)
    private val _progress = MutableStateFlow(0f)
    val progress: StateFlow<Float> = _progress.asStateFlow()

    private fun checkCurrentStatus(): ModelStatus {
        if (!modelFile.exists()) return ModelStatus.NOT_DOWNLOADED
        if (modelFile.length() < MODEL_SIZE_BYTES * 0.9) return ModelStatus.CORRUPTED
        return ModelStatus.DOWNLOADED
    }

    fun getModelPath(): String? {
        return if (_status.value == ModelStatus.DOWNLOADED) modelFile.absolutePath else null
    }

    // 端末の空き容量チェック
    fun getAvailableSpace(): Long {
        val stat = StatFs(context.filesDir.absolutePath)
        return stat.availableBytes
    }

    fun isEnoughSpace(): Boolean {
        return getAvailableSpace() > MODEL_SIZE_BYTES * 1.2  // 20%余裕
    }

    // ダウンロード実行（WorkManagerから呼ばれる）
    suspend fun downloadModel(onProgress: (Float) -> Unit): Boolean {
        if (!modelDir.exists()) modelDir.mkdirs()

        _status.value = ModelStatus.DOWNLOADING

        return try {
            val url = URL(MODEL_URL)
            val connection = url.openConnection() as HttpURLConnection
            connection.connectTimeout = 30_000
            connection.readTimeout = 30_000
            connection.connect()

            val totalBytes = connection.contentLengthLong
            var downloadedBytes = 0L

            connection.inputStream.use { input ->
                FileOutputStream(modelFile).use { output ->
                    val buffer = ByteArray(8192)
                    var bytesRead: Int
                    while (input.read(buffer).also { bytesRead = it } != -1) {
                        output.write(buffer, 0, bytesRead)
                        downloadedBytes += bytesRead
                        val progress = if (totalBytes > 0) downloadedBytes.toFloat() / totalBytes else 0f
                        _progress.value = progress
                        onProgress(progress)
                    }
                }
            }

            _status.value = ModelStatus.DOWNLOADED
            true
        } catch (e: Exception) {
            _status.value = ModelStatus.DOWNLOAD_FAILED
            if (modelFile.exists()) modelFile.delete()
            false
        }
    }

    // モデル削除（設定画面から呼べるように）
    fun deleteModel() {
        if (modelFile.exists()) modelFile.delete()
        _status.value = ModelStatus.NOT_DOWNLOADED
        _progress.value = 0f
    }
}

class GeminiNanoEngine(private val context: Context) : AiEngineInterface {

    private var generativeModel: GenerativeModel? = null

    override suspend fun generate(
        systemPrompt: String,
        userMessage: String,
        maxTokens: Int,
        temperature: Float
    ): String {
        val model = generativeModel ?: Generation.getClient().also {
            generativeModel = it
        }

            generateContentRequest(TextPart(prompt)) {

    }

            status == FeatureStatus.AVAILABLE

        }

class Qwen3LocalEngine(private val modelManager: ModelManager) : AiEngineInterface {

    // 既存の AiEngine クラスの内容をここに移動
    // load() / unload() / generate() をそのまま使用

    private var model: LlamaModel? = null

    override suspend fun generate(
        systemPrompt: String,
        userMessage: String,
        maxTokens: Int,
        temperature: Float
    ): String = withContext(Dispatchers.Default) {
        if (model == null) load()
        val m = model ?: throw IllegalStateException("モデルが読み込まれていません")

        val prompt = """<|im_start|>system
$systemPrompt<|im_end|>
<|im_start|>user
$userMessage<|im_end|>
<|im_start|>assistant
"""
        val inferParams = InferenceParameters()
            .setTemperature(temperature)
            .setNPredict(maxTokens)
            .setStopStrings("<|im_end|>")

        val sb = StringBuilder()
        m.generate(prompt, inferParams) { token -> sb.append(token) }
        sb.toString().trim()
    }

    private fun load(): Boolean {
        val path = modelManager.getModelPath() ?: return false
        return try {
            val params = ModelParameters()
                .setModelFilePath(path)
                .setNGpuLayers(0)
                .setNCtx(4096)
                .setNThreads(4)
            model = LlamaModel(params)
            true
        } catch (e: Exception) {
            model = null
            false
        }
    }

    override fun isAvailable(): Boolean =
        modelManager.status.value == ModelManager.ModelStatus.DOWNLOADED

    override fun getDisplayName() = "Qwen3 1.7B（ローカル）"

    override fun getStatusDescription(): String {
        return when (modelManager.status.value) {
            ModelManager.ModelStatus.DOWNLOADED -> "✅ 利用可能"
            ModelManager.ModelStatus.DOWNLOADING -> "⏳ ダウンロード中"
            ModelManager.ModelStatus.NOT_DOWNLOADED -> "📦 要ダウンロード (1.1GB)"
            ModelManager.ModelStatus.DOWNLOAD_FAILED -> "❌ ダウンロード失敗"
            ModelManager.ModelStatus.CORRUPTED -> "⚠ ファイル破損（再DL必要）"
        }
    }
}

class OpenAiApiEngine(private val dataStore: DataStore<Preferences>) : AiEngineInterface {

    companion object {
        val API_KEY = stringPreferencesKey("openai_api_key")
        val API_MODEL = stringPreferencesKey("openai_model")
        private const val DEFAULT_MODEL = "gpt-4o-mini"
        private const val BASE_URL = "https://api.openai.com/v1/chat/completions"
    }

    override suspend fun generate(
        systemPrompt: String,
        userMessage: String,
        maxTokens: Int,
        temperature: Float
    ): String {
        val prefs = dataStore.data.first()
        val apiKey = prefs[API_KEY] ?: throw IllegalStateException("APIキーが設定されていません")
        val model = prefs[API_MODEL] ?: DEFAULT_MODEL

        val client = OkHttpClient()
        val json = JSONObject().apply {
            put("model", model)
            put("max_tokens", maxTokens)
            put("temperature", temperature)
            put("messages", JSONArray().apply {
                put(JSONObject().apply {
                    put("role", "system")
                    put("content", systemPrompt)
                })
                put(JSONObject().apply {
                    put("role", "user")
                    put("content", userMessage)
                })
            })
        }

        val request = Request.Builder()
            .url(BASE_URL)
            .addHeader("Authorization", "Bearer $apiKey")
            .addHeader("Content-Type", "application/json")
            .post(json.toString().toRequestBody("application/json".toMediaType()))
            .build()

        return withContext(Dispatchers.IO) {
            val response = client.newCall(request).execute()
            val body = response.body?.string() ?: ""
            val result = JSONObject(body)
            result.getJSONArray("choices")
                .getJSONObject(0)
                .getJSONObject("message")
                .getString("content")
                .trim()
        }
    }

    override fun isAvailable(): Boolean {
        return runBlocking {
            val prefs = dataStore.data.first()
            prefs[API_KEY] != null
        }
    }

    override fun getDisplayName() = "OpenAI API（クラウド）"

    override fun getStatusDescription(): String {
        val hasKey = runBlocking {
            dataStore.data.first()[API_KEY] != null
        }
        return if (hasKey) "✅ APIキー設定済み" else "🔑 APIキー未設定"
    }
}

class CustomApiEngine(private val dataStore: DataStore<Preferences>) : AiEngineInterface {

    companion object {
        val ENDPOINT_URL = stringPreferencesKey("custom_api_endpoint")
        val API_KEY = stringPreferencesKey("custom_api_key")  // 任意
        val MODEL_NAME = stringPreferencesKey("custom_api_model")
        private const val DEFAULT_MODEL = "default"
    }

    // OpenAiApiEngine と同じHTTPリクエスト形式
    // ただし BASE_URL を DataStore の ENDPOINT_URL に差し替える
    // API_KEY が空でも動作する（ローカルサーバー用）

    override suspend fun generate(
        systemPrompt: String,
        userMessage: String,
        maxTokens: Int,
        temperature: Float
    ): String {
        val prefs = dataStore.data.first()
        val endpoint = prefs[ENDPOINT_URL]
            ?: throw IllegalStateException("エンドポイントが設定されていません")
        val apiKey = prefs[API_KEY]
        val model = prefs[MODEL_NAME] ?: DEFAULT_MODEL

        val client = OkHttpClient()
        val json = JSONObject().apply {
            put("model", model)
            put("max_tokens", maxTokens)
            put("temperature", temperature)
            put("messages", JSONArray().apply {
                put(JSONObject().apply {
                    put("role", "system")
                    put("content", systemPrompt)
                })
                put(JSONObject().apply {
                    put("role", "user")
                    put("content", userMessage)
                })
            })
        }

        val requestBuilder = Request.Builder()
            .url(endpoint)
            .addHeader("Content-Type", "application/json")
            .post(json.toString().toRequestBody("application/json".toMediaType()))

        if (!apiKey.isNullOrBlank()) {
            requestBuilder.addHeader("Authorization", "Bearer $apiKey")
        }

        return withContext(Dispatchers.IO) {
            val response = client.newCall(requestBuilder.build()).execute()
            val body = response.body?.string() ?: ""
            val result = JSONObject(body)
            result.getJSONArray("choices")
                .getJSONObject(0)
                .getJSONObject("message")
                .getString("content")
                .trim()
        }
    }

    override fun isAvailable(): Boolean {
        return runBlocking {
            dataStore.data.first()[ENDPOINT_URL] != null
        }
    }

    override fun getDisplayName() = "カスタムAPI（OpenAI互換）"

    override fun getStatusDescription(): String {
        val hasEndpoint = runBlocking {
            dataStore.data.first()[ENDPOINT_URL] != null
        }
        return if (hasEndpoint) "✅ エンドポイント設定済み" else "⚙ 未設定"
    }
}

// File: app/src/main/java/com/quickmemo/ai/AiEngineManager.kt

class AiEngineManager(
    private val context: Context,
    private val modelManager: ModelManager,
    private val dataStore: DataStore<Preferences>
) {
    companion object {
        val SELECTED_ENGINE = stringPreferencesKey("selected_ai_engine")
    }

    private val engines = mapOf(
        AiEngineType.GEMINI_NANO to GeminiNanoEngine(context),
        AiEngineType.QWEN3_LOCAL to Qwen3LocalEngine(modelManager),
        AiEngineType.OPENAI_API to OpenAiApiEngine(dataStore),
        AiEngineType.CUSTOM_API to CustomApiEngine(dataStore)
    )

    fun getEngine(type: AiEngineType): AiEngineInterface =
        engines[type] ?: throw IllegalArgumentException("未知のエンジン: $type")

    fun getAllEngines(): Map<AiEngineType, AiEngineInterface> = engines

    suspend fun getSelectedEngine(): AiEngineInterface {
        val prefs = dataStore.data.first()
        val typeName = prefs[SELECTED_ENGINE] ?: AiEngineType.QWEN3_LOCAL.name
        val type = AiEngineType.valueOf(typeName)
        return getEngine(type)
    }

    suspend fun setSelectedEngine(type: AiEngineType) {
        dataStore.edit { it[SELECTED_ENGINE] = type.name }
    }
}

Copyclass OcrProcessor {

    private val recognizer = TextRecognition.getClient(
        JapaneseTextRecognizerOptions.Builder().build()
    )

    suspend fun recognizeFromUri(context: Context, uri: Uri): OcrResult {
        return suspendCoroutine { continuation ->
            try {
                val image = InputImage.fromFilePath(context, uri)
                recognizer.process(image)
                    .addOnSuccessListener { visionText ->
                        continuation.resume(OcrResult.Success(visionText.text))
                    }
                    .addOnFailureListener { e ->
                        continuation.resume(OcrResult.Error(e.message ?: "認識に失敗しました"))
                    }
            } catch (e: Exception) {
                continuation.resume(OcrResult.Error(e.message ?: "画像の読み込みに失敗しました"))
            }
        }
    }

    suspend fun recognizeFromBitmap(bitmap: Bitmap): OcrResult {
        return suspendCoroutine { continuation ->
            val image = InputImage.fromBitmap(bitmap, 0)
            recognizer.process(image)
                .addOnSuccessListener { visionText ->
                    continuation.resume(OcrResult.Success(visionText.text))
                }
                .addOnFailureListener { e ->
                    continuation.resume(OcrResult.Error(e.message ?: "認識に失敗しました"))
                }
        }
    }
}

// File: app/src/main/java/com/quickmemo/translation/TranslationManager.kt

Copyclass TranslationManager {

    private var jaToEnTranslator: Translator? = null
    private var enToJaTranslator: Translator? = null

    private val _downloadStatus = MutableStateFlow<TranslationDownloadStatus>(
        TranslationDownloadStatus.NotChecked
    )
    val downloadStatus: StateFlow<TranslationDownloadStatus> = _downloadStatus.asStateFlow()

    sealed class TranslationDownloadStatus {
        object NotChecked : TranslationDownloadStatus()
        object Downloading : TranslationDownloadStatus()
        object Ready : TranslationDownloadStatus()
        data class Error(val message: String) : TranslationDownloadStatus()
    }

    // 翻訳モデルのダウンロード（各約30MB、初回のみ）
    suspend fun ensureModelsDownloaded() {
        _downloadStatus.value = TranslationDownloadStatus.Downloading

        try {
            val jaToEnOptions = TranslatorOptions.Builder()
                .setSourceLanguage(TranslateLanguage.JAPANESE)
                .setTargetLanguage(TranslateLanguage.ENGLISH)
                .build()

            val enToJaOptions = TranslatorOptions.Builder()
                .setSourceLanguage(TranslateLanguage.ENGLISH)
                .setTargetLanguage(TranslateLanguage.JAPANESE)
                .build()

            jaToEnTranslator = Translation.getClient(jaToEnOptions)
            enToJaTranslator = Translation.getClient(enToJaOptions)

            // モデルダウンロード
            val conditions = DownloadConditions.Builder()
                .requireWifi()
                .build()

            jaToEnTranslator!!.downloadModelIfNeeded(conditions).await()
            enToJaTranslator!!.downloadModelIfNeeded(conditions).await()

            _downloadStatus.value = TranslationDownloadStatus.Ready
        } catch (e: Exception) {
            _downloadStatus.value = TranslationDownloadStatus.Error(
                e.message ?: "翻訳モデルのダウンロードに失敗しました"
            )
        }
    }

    // 日本語 → 英語
    suspend fun translateJaToEn(text: String): String {
        val translator = jaToEnTranslator
            ?: throw IllegalStateException("翻訳モデルが準備されていません")
        return translator.translate(text).await()
    }

    // 英語 → 日本語
    suspend fun translateEnToJa(text: String): String {
        val translator = enToJaTranslator
            ?: throw IllegalStateException("翻訳モデルが準備されていません")
        return translator.translate(text).await()
    }

    // 自動言語検出して翻訳（日本語→英語、それ以外→日本語）
    suspend fun autoTranslate(text: String): TranslationResult {
        val languageIdentifier = LanguageIdentification.getClient()
        val detectedLang = suspendCoroutine<String> { cont ->
            languageIdentifier.identifyLanguage(text)
                .addOnSuccessListener { lang -> cont.resume(lang) }
                .addOnFailureListener { cont.resume("und") }
        }

        return when (detectedLang) {
            "ja" -> TranslationResult(
                translatedText = translateJaToEn(text),
                sourceLanguage = "日本語",
                targetLanguage = "英語"
            )
            else -> TranslationResult(
                translatedText = translateEnToJa(text),
                sourceLanguage = "英語",
                targetLanguage = "日本語"
            )
        }
    }

    fun close() {
        jaToEnTranslator?.close()
        enToJaTranslator?.close()
    }
}

data class TranslationResult(
    val translatedText: String,
    val sourceLanguage: String,
    val targetLanguage: String
